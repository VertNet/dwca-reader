001package org.gbif.dwc.text;
002
003import org.gbif.dwc.record.Record;
004import org.gbif.dwc.terms.Term;
005
006import java.util.ArrayList;
007import java.util.Collection;
008import java.util.Iterator;
009import java.util.List;
010import java.util.Map;
011import java.util.Set;
012import java.util.SortedMap;
013import java.util.TreeMap;
014
015public class StarRecord implements Iterable<Record> {
016
017  private Record core;
018  private final SortedMap<String, List<Record>> extensions;
019
020  public StarRecord(Collection<String> extensions) {
021    this.extensions = new TreeMap<String, List<Record>>();
022    for (String rowType : extensions) {
023      this.extensions.put(rowType, new ArrayList<Record>());
024    }
025  }
026
027  public void addRecord(String rowType, Record record) {
028    if (!extensions.containsKey(rowType)) {
029      throw new IllegalArgumentException("RowType not supported");
030    }
031    extensions.get(rowType).add(record);
032  }
033
034  /**
035   * @return the core record
036   */
037  public Record core() {
038    return core;
039  }
040
041  /**
042   * Retrieves all extension records of a specific extension.
043   * If the requested extension is not mapped null will be returned.
044   *
045   * @param rowType the Term representing the rowType
046   *
047   * @return possibly empty list of extension record or null if extension is not mapped at all
048   */
049  public List<Record> extension(Term rowType) {
050    return extension(rowType.qualifiedName());
051  }
052
053  /**
054   * Retrieves all extension records of a specific extension.
055   * If the requested extension is not mapped null will be returned.
056   *
057   * @param rowType the exact qualified name of the rowType, e.g. http://rs.gbif.org/terms/1.0/VernacularName
058   *
059   * @return possibly empty list of extension record or null if extension is not mapped at all
060   */
061  public List<Record> extension(String rowType) {
062    return extensions.get(rowType);
063  }
064
065  /**
066   * Retrieves all extension records related to the core record across all extensions as a map.
067   */
068  public Map<String, List<Record>> extensions() {
069    return extensions;
070  }
071
072  /**
073   * Get a new iterator over all extension records, no matter to which extension they belong.
074   *
075   * @return the ExtensionRecord iterator, reusing the same instance for each call
076   */
077  public Iterator<Record> iterator() {
078    List<Record> records = new ArrayList<Record>();
079    for (List<Record> recs : extensions.values()) {
080      records.addAll(recs);
081    }
082    return records.iterator();
083  }
084
085  public void newCoreRecord(Record core) {
086    this.core = core;
087    for (List<Record> lists : extensions.values()) {
088      lists.clear();
089    }
090  }
091
092  /**
093   * @return set of extension rowTypes associated with this star record
094   */
095  public Set<String> rowTypes() {
096    return extensions.keySet();
097  }
098
099  /**
100   * @return the number of associated extension records across all rowTypes
101   */
102  public int size() {
103    int x = 0;
104    for (List<Record> recs : extensions.values()) {
105      x += recs.size();
106    }
107    return x;
108  }
109}
