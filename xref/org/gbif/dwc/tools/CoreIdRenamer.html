42  /**
43   * Tool that takes a dwca file and renames its core ids according to some id generator.
44   * All extension records and ID terms pointing to the core ID are renamed accordingly to preserve the archive
45   * integrity.
46   */
47  public class CoreIdRenamer {
48  
49    private final IdGenerator idGen;
50  
51    public CoreIdRenamer() {
52      idGen = new IntSequenceGenerator(1);
53    }
54  
55    public CoreIdRenamer(IdGenerator idGen) {
56      this.idGen = idGen;
57    }
58  
59    public CoreIdRenamer(int firstId) {
60      idGen = new IntSequenceGenerator(firstId);
61    }
62  
63    public void rename(File dwca) throws IOException {
64      rename(ArchiveFactory.openArchive(dwca));
65    }
66  
67    public void rename(Archive arch) throws IOException {
68      ArchiveField id = arch.getCore().getId();
69      TermFactory fact = TermFactory.instance();
70      Term rowType = fact.findTerm(arch.getCore().getRowType());
71  
72      // detect dependent terms with ids to be renamed accordingly
73      List<Term> relatedTerms = new ArrayList<Term>();
74      if (DwcTerm.Taxon == rowType) {
75        // potentially 3 foreign keys with values pointing to the coreID
76        if (arch.getCore().hasTerm(DwcTerm.parentNameUsageID)) {
77          relatedTerms.add(DwcTerm.parentNameUsageID);
78        }
79        if (arch.getCore().hasTerm(DwcTerm.acceptedNameUsageID)) {
80          relatedTerms.add(DwcTerm.acceptedNameUsageID);
81        }
82        if (arch.getCore().hasTerm(DwcTerm.originalNameUsageID)) {
83          relatedTerms.add(DwcTerm.originalNameUsageID);
84        }
85      } else if (DwcTerm.Occurrence == rowType) {
86        // none as far as we know
87      } else {
88        throw new IllegalStateException("Unknown core rowtype - cannot rename the archive ids");
89      }
90  
91      // map old to new ids and store them in memory
92      Map<String, String> old2new = new HashMap<String, String>();
93      ClosableIterator<DarwinCoreRecord> iter = arch.iteratorDwc();
94      System.out.println("Mapping record ids ...");
95      while (iter.hasNext()) {
96        DarwinCoreRecord rec = iter.next();
97        old2new.put(rec.id(), idGen.nextId());
98      }
99      System.out.println("Mapped " + old2new.size() + " record ids. Rewrite files ...");
100 
101     // rename all files
102     rewriteFile(arch.getCore(), old2new, relatedTerms);
103     for (ArchiveFile af : arch.getExtensions()) {
104       rewriteFile(arch.getCore(), old2new, relatedTerms);
105     }
106   }

108   private void rewriteFile(ArchiveFile original, Map<String, String> old2new, List<Term> relatedTerms)
109     throws IOException {
110     File newFile = new File(original.getLocationFile().getParent(), original.getLocationFile().getName() + "-NEW");
111     Writer writer =
112       new BufferedWriter(new OutputStreamWriter(new FileOutputStream(newFile, false), original.getEncoding()));
113 
114     List<Integer> renameCols = new ArrayList<Integer>();
115     renameCols.add(original.getId().getIndex());
116     for (Term t : relatedTerms) {
117       if (original.hasTerm(t)) {
118         renameCols.add(original.getField(t).getIndex());
119       }
120     }
121     // iterate thru rows
122     CSVReader reader = original.getCSVReader();
123     final String quote = original.getFieldsEnclosedBy() == null ? "" : original.getFieldsEnclosedBy().toString();
124     final String quoteDelimQuote = quote + original.getFieldsTerminatedBy() + quote;
125 
126     // headers
127     for (int hr = reader.headerRows; hr > 0; hr--) {
128       // write row with given delimter, quotation and line ending
129       String rowAsString = quote + StringUtils.join(reader.header, quoteDelimQuote) + quote;
130       writer.write(rowAsString);
131       writer.write('\n');
132     }
133     // records
134     for (String[] row : reader) {
135       for (Integer col : renameCols) {
136         row[col] = old2new.get(row[col]);
137       }
138       // write row with given delimter, quotation and line ending
139       String rowAsString = quote + StringUtils.join(row, quoteDelimQuote) + quote;
140       writer.write(rowAsString);
141       writer.write('\n');
142     }
143     writer.flush();
144     writer.close();
145   }
