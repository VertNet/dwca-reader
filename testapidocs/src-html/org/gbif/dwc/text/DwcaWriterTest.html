076  @Test
077  public void testHeaderWriting() throws Exception {
078    File dwcaDir = FileUtils.createTempDir();
079    dwcaDir.deleteOnExit();
080    LOG.info("Test archive writer in {}", dwcaDir.getAbsolutePath());
081
082    DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, dwcaDir, true);
083
084    writer.newRecord("dummy1");
085    writer.addCoreColumn(DwcTerm.parentNameUsageID, null);
086    writer.addCoreColumn(DwcTerm.acceptedNameUsageID, null);
087    writer.addCoreColumn(DwcTerm.scientificName, null);
088    writer.addCoreColumn(GbifTerm.canonicalName, null);
089    writer.addCoreColumn(DwcTerm.taxonRank, "species");
090    writer.addCoreColumn(DwcTerm.taxonomicStatus, null);
091    writer.addCoreColumn(DwcTerm.kingdom, null);
092    writer.addCoreColumn(DwcTerm.phylum, null);
093    writer.addCoreColumn(DwcTerm.class_, null);
094    writer.addCoreColumn(DwcTerm.order, null);
095    writer.addCoreColumn(DwcTerm.family, null);
096    writer.addCoreColumn(GbifTerm.depth, null);
097    writer.addCoreColumn(GbifTerm.depthAccuracy, null);
098
099    writer.newRecord("dummy2");
100    writer.addCoreColumn(DwcTerm.kingdom, "Plantae");
101    writer.addCoreColumn(DwcTerm.phylum, null);
102    writer.addCoreColumn(DwcTerm.class_, null);
103    writer.addCoreColumn(DwcTerm.order, null);
104    writer.addCoreColumn(DwcTerm.family, "Asteraceae");
105
106    writer.newRecord("dummy3");
107    writer.addCoreColumn(GbifTerm.depth, "2");
108    writer.addCoreColumn(GbifTerm.depthAccuracy, "1");
109
110    // define extension columns
111    Map<Term, String> eData = Maps.newHashMap();
112    // distributions
113    eData.put(DwcTerm.locality, "locality1");
114    eData.put(DwcTerm.occurrenceStatus, "present");
115    eData.put(DwcTerm.establishmentMeans, "alien");
116    writer.addExtensionRecord(GbifTerm.Distribution, eData);
117
118    eData.put(DwcTerm.locality, "locality2");
119    writer.addExtensionRecord(GbifTerm.Distribution, eData);
120
121    writer.close();
122
123    File cf = new File(dwcaDir, writer.getDataFiles().get(DwcTerm.Taxon));
124    File df = new File(dwcaDir, writer.getDataFiles().get(GbifTerm.Distribution));
125
126    // check if taxon file contains headers
127    String[] headers = getFirstRow(cf);
128    LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
129    assertEquals(14, headers.length);
130    assertEquals("taxonID", headers[0]);
131    assertEquals("parentNameUsageID", headers[1]);
132    assertEquals("kingdom", headers[7]);
133
134    // check if extension file contains headers
135    headers = getFirstRow(df);
136    LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
137    assertEquals(4, headers.length);
138    assertEquals("taxonID", headers[0]);
139  }

141  private String[] getFirstRow(File f) throws IOException {
142    BufferedReader r = FileUtils.getUtf8Reader(f);
143    String firstRow = r.readLine();
144    return firstRow.split("\t");
145  }

147  @Test
148  public void testRoundtrip() throws Exception {
149    try {
150      TermFactory termFactory = TermFactory.instance();
151
152      // read taxon archive
153      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
154      assertEquals(1, arch.getExtensions().size());
155      int coreRecords = 0;
156      int allRecords = 0;
157
158      // write taxon archive
159      File tempArch = FileUtils.createTempDir();
160      tempArch.deleteOnExit();
161      System.out.println("Writing temporary test archive to " + tempArch.getAbsolutePath());
162      DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, tempArch);
163      for (StarRecord rec : arch) {
164        // core
165        coreRecords++;
166        allRecords += rec.size();
167
168        writer.newRecord(rec.core().id());
169        for (Term term : arch.getCore().getTerms()) {
170          writer.addCoreColumn(term, rec.core().value(term));
171        }
172        // extensions
173        for (String rt : rec.extensions().keySet()) {
174          Term rowType = termFactory.findTerm(rt);
175          ArchiveFile af = arch.getExtension(rowType);
176          // iterate over records for one extension
177          for (Record row : rec.extension(rt)) {
178            writer.addExtensionRecord(rowType, DwcaWriter.recordToMap(row, af));
179          }
180        }
181      }
182      writer.close();
183
184      // reread and compare
185      Archive arch2 = ArchiveFactory.openArchive(tempArch);
186      int coreRecords2 = 0;
187      int allRecords2 = 0;
188      for (StarRecord rec : arch2) {
189        // core
190        coreRecords2++;
191        allRecords2 += rec.size();
192      }
193
194      // compare
195      assertEquals(coreRecords, coreRecords2);
196      assertEquals(allRecords, allRecords2);
197
198
199    } catch (Exception e) {
200      e.printStackTrace();
201      fail();
202    }
203  }
