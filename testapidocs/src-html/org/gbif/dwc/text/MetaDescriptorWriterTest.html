001package org.gbif.dwc.text;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.utils.file.FileUtils;
005
006import java.io.File;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011import static org.junit.Assert.assertNotNull;
012import static org.junit.Assert.assertNull;
013import static org.junit.Assert.assertTrue;
014import static org.junit.Assert.fail;
015
016public class MetaDescriptorWriterTest {
017
018    @Test
019    public void testRoundtrip() {
020        try {
021            // read archive
022            Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
023            assertNotNull(arch);
024            assertNotNull(arch.getCore());
025            assertTrue(arch.getCore().getId().getIndex() == 0);
026            assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
027            assertEquals(2, arch.getExtensions().size());
028            assertEquals("\t", arch.getCore().getFieldsTerminatedBy());
029            assertNull(arch.getCore().getField(DwcTerm.scientificName).getDelimitedBy());
030            assertEquals(";", arch.getCore().getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
031
032            // write meta.xml
033            File temp = File.createTempFile("meta", ".xml");
034            temp.deleteOnExit();
035            System.out.println("Writing temporary test eml file to " + temp.getAbsolutePath());
036            MetaDescriptorWriter.writeMetaFile(temp, arch);
037
038            Archive arch2 = ArchiveFactory.openArchive(temp);
039            // core props
040            ArchiveFile core = arch2.getCore();
041            assertNotNull(core);
042            assertNotNull(core.getId());
043            assertTrue(core.hasTerm(DwcTerm.scientificName));
044            assertEquals("DarwinCore.txt", core.getLocation());
045            assertEquals("\t", core.getFieldsTerminatedBy());
046            assertNull(arch.getCore().getField(DwcTerm.scientificName).getDelimitedBy());
047            assertEquals(";", arch.getCore().getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
048            for (ArchiveField f : arch.getCore().getFields().values()) {
049                assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
050                assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
051            }
052
053            // extensions props
054            assertEquals(2, arch2.getExtensions().size());
055            ArchiveFile ext = arch2.getExtensions().iterator().next();
056            assertEquals("VernacularName.txt", ext.getLocation());
057            assertEquals(2, ext.getFields().size());
058
059        } catch (Exception e) {
060            e.printStackTrace();
061            fail();
062        }
063    }
064
065    @Test
066    public void testRoundtripQuotes() {
067        try {
068            // read archive
069            Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("meta_quot.xml"));
070            assertNotNull(arch);
071            assertNotNull(arch.getCore());
072            assertNotNull(arch.getCore().getId());
073            assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
074            assertEquals(1, arch.getExtensions().size());
075
076            // write meta.xml
077            File temp = File.createTempFile("meta", ".xml");
078            temp.deleteOnExit();
079            //      System.out.println("Writing temporary test meta.xml file to " + temp.getAbsolutePath());
080            MetaDescriptorWriter.writeMetaFile(temp, arch);
081
082            Archive arch2 = ArchiveFactory.openArchive(temp);
083            // core props
084            ArchiveFile core = arch2.getCore();
085            assertNotNull(core);
086            assertNotNull(core.getId());
087            assertTrue(core.hasTerm(DwcTerm.scientificName));
088            assertEquals("test", core.getLocation());
089            for (ArchiveField f : arch.getCore().getFields().values()) {
090                assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
091                assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
092            }
093
094            // extensions props
095            assertEquals(1, arch2.getExtensions().size());
096            ArchiveFile ext = arch2.getExtensions().iterator().next();
097            assertEquals("test2", ext.getLocation());
098            assertEquals(2, ext.getFields().size());
099
100        } catch (Exception e) {
101            e.printStackTrace();
102            fail();
103        }
104    }
105}
