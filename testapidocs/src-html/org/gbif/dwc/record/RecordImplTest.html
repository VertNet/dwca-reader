035  @Test
036  public void testReplaceNull() {
037    RecordImpl r = new RecordImpl(null, (Map<Term, ArchiveField>) null, null, true);
038    assertNull(r.replaceNull(null));
039    assertNull(r.replaceNull(""));
040    assertNull(r.replaceNull(" "));
041    assertNull(r.replaceNull("    "));
042    assertNull(r.replaceNull("null"));
043    assertNull(r.replaceNull("NULL"));
044    assertNull(r.replaceNull("\\N"));
045    assertNull(r.replaceNull(" Null  "));
046    assertEquals("n", r.replaceNull("n"));
047    assertEquals("n ", r.replaceNull("n "));
048    assertEquals(" - ", r.replaceNull(" - "));
049    assertEquals("nulle", r.replaceNull("nulle"));
050    assertEquals("n ull", r.replaceNull("n ull"));
051  }
052
053  @Test
054  public void testGetFullScientificName() {
055    ArchiveField id = new ArchiveField(0, DwcTerm.taxonID, null, null);
056    Set<ArchiveField> fields = new HashSet<ArchiveField>();
057    fields.add(new ArchiveField(1, DwcTerm.scientificName, null, null));
058    fields.add(new ArchiveField(2, DwcTerm.scientificNameAuthorship, null, null));
059    fields.add(new ArchiveField(null, DwcTerm.kingdom, "Plantae", null));
060    fields.add(new ArchiveField(4, DwcTerm.phylum, null, null));
061    fields.add(new ArchiveField(6, DwcTerm.class_, null, null));
062    fields.add(new ArchiveField(7, DwcTerm.acceptedNameUsage, null, null));
063    RecordImpl rec = new RecordImpl(id, fields, DwcTerm.Taxon.qualifiedName(), true);
064
065    String[] row =
066      {"5432", "Abies alba Mill.", "Mill.", "Harry", "Monocotyledonae", "Bertram", "Pincodiae", "Picea picaea L."};
067    rec.setRow(row);
068
069    assertEquals(row[1], rec.value(DwcTerm.scientificName));
070    assertEquals(row[2], rec.value(DwcTerm.scientificNameAuthorship));
071    assertEquals("Plantae", rec.value(DwcTerm.kingdom));
072    assertEquals(row[4], rec.value(DwcTerm.phylum));
073    assertEquals(row[6], rec.value(DwcTerm.class_));
074    assertEquals(row[6], rec.value(DwcTerm.class_.qualifiedName()));
075    assertNull(rec.value(DwcTerm.order));
076    assertEquals(row[7], rec.value(DwcTerm.acceptedNameUsage));
077
078    assertEquals(row[0], rec.id());
079    assertEquals(DwcTerm.Taxon.qualifiedName(), rec.rowType());
080  }
081
082  @Test
083  public void testDefaultValue() {
084    final String DATASET = "ITIS";
085    ArchiveField id = new ArchiveField(0, DwcTerm.taxonID, null, null);
086    Set<ArchiveField> fields = new HashSet<ArchiveField>();
087    fields.add(new ArchiveField(1, DwcTerm.datasetName, DATASET, null));
088    RecordImpl rec = new RecordImpl(id, fields, DwcTerm.Taxon.qualifiedName(), true);
089
090    String[] row = {"5432", "IPNI"};
091    rec.setRow(row);
092    assertEquals(row[1], rec.value(DwcTerm.datasetName));
093
094    row = new String[] {"5432", null};
095    rec.setRow(row);
096    assertEquals(DATASET, rec.value(DwcTerm.datasetName));
097
098    row = new String[] {"5432", ""};
099    rec.setRow(row);
100    assertEquals(DATASET, rec.value(DwcTerm.datasetName));
101  }
