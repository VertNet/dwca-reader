35    @Test
36    public void testReplaceNull() {
37      RecordImpl r = new RecordImpl(null, (Map<Term, ArchiveField>) null, null, true);
38      assertNull(r.replaceNull(null));
39      assertNull(r.replaceNull(""));
40      assertNull(r.replaceNull(" "));
41      assertNull(r.replaceNull("    "));
42      assertNull(r.replaceNull("null"));
43      assertNull(r.replaceNull("NULL"));
44      assertNull(r.replaceNull("\\N"));
45      assertNull(r.replaceNull(" Null  "));
46      assertEquals("n", r.replaceNull("n"));
47      assertEquals("n ", r.replaceNull("n "));
48      assertEquals(" - ", r.replaceNull(" - "));
49      assertEquals("nulle", r.replaceNull("nulle"));
50      assertEquals("n ull", r.replaceNull("n ull"));
51    }

53    @Test
54    public void testGetFullScientificName() {
55      ArchiveField id = new ArchiveField(0, DwcTerm.taxonID, null, null);
56      Set<ArchiveField> fields = new HashSet<ArchiveField>();
57      fields.add(new ArchiveField(1, DwcTerm.scientificName, null, null));
58      fields.add(new ArchiveField(2, DwcTerm.scientificNameAuthorship, null, null));
59      fields.add(new ArchiveField(null, DwcTerm.kingdom, "Plantae", null));
60      fields.add(new ArchiveField(4, DwcTerm.phylum, null, null));
61      fields.add(new ArchiveField(6, DwcTerm.class_, null, null));
62      fields.add(new ArchiveField(7, DwcTerm.acceptedNameUsage, null, null));
63      RecordImpl rec = new RecordImpl(id, fields, DwcTerm.Taxon.qualifiedName(), true);
64  
65      String[] row =
66        {"5432", "Abies alba Mill.", "Mill.", "Harry", "Monocotyledonae", "Bertram", "Pincodiae", "Picea picaea L."};
67      rec.setRow(row);
68  
69      assertEquals(row[1], rec.value(DwcTerm.scientificName));
70      assertEquals(row[2], rec.value(DwcTerm.scientificNameAuthorship));
71      assertEquals("Plantae", rec.value(DwcTerm.kingdom));
72      assertEquals(row[4], rec.value(DwcTerm.phylum));
73      assertEquals(row[6], rec.value(DwcTerm.class_));
74      assertEquals(row[6], rec.value(DwcTerm.class_.qualifiedName()));
75      assertNull(rec.value(DwcTerm.order));
76      assertEquals(row[7], rec.value(DwcTerm.acceptedNameUsage));
77  
78      assertEquals(row[0], rec.id());
79      assertEquals(DwcTerm.Taxon.qualifiedName(), rec.rowType());
80    }
