33    @Test
34    public void detectCsvAlwaysQuoted() throws IOException {
35      String[] files = {"csv_always_quoted.csv"};
36      for (String fn : files) {
37        File source = FileUtils.getClasspathFile(fn);
38        CSVReader reader = CSVReaderFactory.build(source);
39        assertEquals(",", reader.delimiter);
40        assertEquals(new Character('"'), reader.quoteChar);
41        assertEquals(1, reader.headerRows);
42        reader.close();
43      }
44    }

46    /**
47     * As CSV files with rare optional quotes are hard to detect but cause problems
48     * we prefer to default to the " quotation in case comma seperated files are used.

51    @Test
